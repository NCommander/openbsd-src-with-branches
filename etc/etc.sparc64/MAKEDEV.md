	_recurse std lkm random
	_recurse fd0 rd0 wd0 wd1 wd2 wd3 bpf0

_std(2,3,76,7,16)
	M mouse		c 13 0 666
	M fb		c 22 0 666

	;;

mouse*)name=${i##mouse-}
	if [ ! -c $name ]; then
		$0 $name	# make the appropriate device
	fi
	RMlist="$RMlist mouse"
	MKlist="$MKlist;ln -s $name mouse";;

magma*)
	case $U in
	0)	offset=0  nam=m;;
	1)	offset=16 nam=n;;
	2)	offset=32 nam=o;;
	*)	echo "bad unit for $i: $U"; exit 127;;
	esac
	offset=Mult($U,64)
	n=0
	while [ $n -lt 16 ]
	do
		name=${nam}`hex $n`
		M tty$name c 71 Add($offset,$n) 660 dialer uucp
		n=Add($n,1)
	done
	M bpp${nam}0 c 72 Add($offset,0) 600 wheel root
	M bpp${nam}1 c 72 Add($offset,1) 600 wheel root
	;;
